#include <stdio.h>
#include <string.h>


int findWin(float points[], int n)
{
    int ind = 0;
    for (int i = 0; i < n; i++)
    {
        if (points[i] > points[ind])
        {
            ind = i;
        }
    }
    return ind;
}
void readArray(float a[], int n)
{
    for (int i = 0; i < n; i++)
    {
        scanf("%f",&a[i]);
    }
}

float sumW(float a[], int m)
{
    float sum = a[0], min = a[0], max = a[0];
    for (int z = 1; z < m; z++)
    {
        sum += a[z];
        if (a[z] < min)
            min = a[z];
        if (a[z] > max)
            max = a[z];
    }
    return sum - (max + min);
}


int main()
{
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    char names[n][20];
    float points[n];
    memset(points,0x0,sizeof(points));
    for (int i = 0; i < k; i++)
    {
        for (int j = 0; j < n; j++)
        {
            scanf("%s",names[j]);
            float a[m];
            readArray(a,m);
            points[j] += sumW(a,m);
        }
    }
    int ind = findWin(points,n);
    printf("%s",names[ind]);
    return 0;
}
