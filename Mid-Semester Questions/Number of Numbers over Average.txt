Find the number of numbers which are greater than their average

//rbitri
#include <stdio.h>
#include <stdlib.h>

int getN()
{
    int n;
    scanf("%d", &n);
    return n;
}

void getArray(int n, int arr[])
{
    for(int i=0; i<n; i++)
    {
        arr[i]=getN();
    }
}

int Sum(int n, int arr[])
{
    int sum=0;
    for(int i=0; i<n; i++)
    {
        sum+=arr[i];
    }
    return sum;
}

double calcAvg(int n, int sum)
{
    double avg;
    avg=(double)sum/n;
    return avg;
}

int checkNumbers(int n, int arr[], double avg)
{
    int k=0;
    for(int i=0; i<n; i++)
    {
        if(arr[i]>avg)
            k++;
    }

    return k;
}

void showData(int k)
{
    printf("%d", k);
}
int main()
{
    int n, arr[10000];
    int sum, k;
    double avg;
    n=getN();
    getArray(n, arr);
    sum=Sum(n, arr);
    avg=calcAvg(n, sum);
    k=checkNumbers(n, arr, avg);
    showData(k);
    return 0;
}
